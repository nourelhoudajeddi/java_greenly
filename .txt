package controllers;

import javafx.fxml.FXML;
import javafx.scene.web.WebView;
import javafx.scene.web.WebEngine;

import java.io.InputStream;
import java.nio.charset.StandardCharsets;
import java.util.Scanner;

public class MapController {

    @FXML
    private WebView mapView;

    @FXML
    public void initialize() {
        WebEngine webEngine = mapView.getEngine();
        // Charger le fichier HTML local
        String html = loadHtmlFile("/map.html"); // Assurez-vous que map.html est dans src/main/resources
        webEngine.loadContent(html);
    }

    private String loadHtmlFile(String path) {
        InputStream inputStream = getClass().getResourceAsStream(path);
        if (inputStream == null) {
            throw new RuntimeException("Fichier HTML introuvable : " + path);
        }

        Scanner scanner = new Scanner(inputStream, StandardCharsets.UTF_8).useDelimiter("\\A");
        return scanner.hasNext() ? scanner.next() : "";
    }
}

    <script>
        function initMap() {
            var userLocation = [48.8566, 2.3522]; // Paris par défaut (en cas d'échec de géolocalisation)

            // Initialisation de la carte
            var map = L.map('map').setView(userLocation, 13);

            // Ajouter la couche OpenStreetMap
            L.tileLayer('https://{s}.tile.openstreetmap.org/{z}/{x}/{y}.png', {
                attribution: '&copy; <a href="https://www.openstreetmap.org/copyright">OpenStreetMap</a> contributors'
            }).addTo(map);

            // Ajouter un marqueur
            L.marker(userLocation).addTo(map)
                .bindPopup("Votre Position")
                .openPopup();
        }
    </script>
</head>
<body onload="initMap()">
<div id="map" style="width:100%; height:100vh;"></div>
</body>
